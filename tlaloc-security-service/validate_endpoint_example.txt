// Validar Token Endpoint
e.POST("/auth/validate", func(c echo.Context) error {
	authHeader := c.Request().Header.Get("Authorization")
	if authHeader == "" {
		return c.JSON(400, dto.ErrorResponse{
			Error:   "missing_token",
			Message: "Authorization header is required",
			Code:    400,
		})
	}

	tokenParts := strings.Split(authHeader, " ")
	if len(tokenParts) != 2 || tokenParts[0] != "Bearer" {
		return c.JSON(400, dto.ErrorResponse{
			Error:   "invalid_token_format",
			Message: "Token must be in 'Bearer <token>' format",
			Code:    400,
		})
	}

	token := tokenParts[1]
	claims, err := authMiddleware.jwtService.ValidateAccessToken(token)
	if err != nil {
		return c.JSON(401, dto.ErrorResponse{
			Error:   "invalid_token",
			Message: "Token is invalid or expired",
			Code:    401,
		})
	}

	return c.JSON(200, map[string]interface{}{
		"valid":   true,
		"user_id": claims.UserID,
		"email":    claims.Email,
		"role":     claims.Role,
		"expires_at": claims.ExpiresAt,
	})
})
